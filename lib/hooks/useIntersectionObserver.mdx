# useNavigationObserver

```
const [stopNavigation, setStopNavigation] = useState(true)

useNavigationObserver({
    shouldStopNavigation: stopNavigation,
    onNavigate: (path) => {
        if (confirm('Exit?')) { // Add condition here
            setStopNavigation(false)
            router.push(path)
        }
    },
})
```
